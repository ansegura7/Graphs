# Main data variables
n_rows = 69
top_country = 10
exp_country = 'Colombia'
exp_data = None
countries_data = dict()
In [6]:
for row in raw_data.iterrows():
    row = row[1]
    country_name = row["country"]
    x = []
    
    if country_name == exp_country:
        for i in range(1, n_rows + 1):
            if pd.notna(row[str(i)]):
                x.append(int(row[str(i)]))
        
        exp_data = np.zeros((n_rows, 2))
        exp_data[:, 0] = x
        exp_data[:, 1] = np.arange(n_rows)
    else:
        temp_data = []
        for i in range(1, n_rows + 1):
            if pd.notna(row[str(i)]):
                x.append(int(row[str(i)]))
        
        if len(x) == n_rows:
            temp_data = np.zeros((n_rows, 2))
            temp_data[:, 0] = x
            temp_data[:, 1] = np.arange(n_rows)
            countries_data[country_name] = temp_data

# Quantify the difference between the two curves using PCM
pcm_result = dict()
for country, num_data in countries_data.items():
    pcm = similaritymeasures.pcm(exp_data, num_data)
    pcm_result[country] = pcm

output = sorted(pcm_result, key=pcm_result.get, reverse=False)[:top_country]
